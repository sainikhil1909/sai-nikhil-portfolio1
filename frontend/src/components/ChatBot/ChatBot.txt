import React, {
useState,
useEffect,
useRef
} from "react";

import {
FaRobot,
FaTimes,
FaPaperPlane
} from "react-icons/fa";

import "./ChatBot.css";

function ChatBot() {

const [open, setOpen] = useState(false);

const [input, setInput] = useState("");

const [visitorName, setVisitorName] = useState(
localStorage.getItem("visitorName") || ""
);

const [nameAsked, setNameAsked] = useState(
localStorage.getItem("visitorName")
? true
: false
);

const [messages, setMessages] = useState(
JSON.parse(
localStorage.getItem("chatMessages")
) || [
{
sender: "bot",
text:
"👋 Welcome to Sai Nikhil's Portfolio.\n\nMay I know your name?"
}
]
);

const messagesEndRef = useRef(null);

/* Auto Scroll */

useEffect(() => {


messagesEndRef.current?.scrollIntoView({
  behavior: "smooth"
});


}, [messages]);

/* Save Messages */

useEffect(() => {


localStorage.setItem(
  "chatMessages",
  JSON.stringify(messages)
);


}, [messages]);

/* Save Visitor Name */

useEffect(() => {


if (visitorName) {

  localStorage.setItem(
    "visitorName",
    visitorName
  );

}


}, [visitorName]);

/* Clear when tab closes */

useEffect(() => {


const clearChat = () => {

  localStorage.removeItem(
    "chatMessages"
  );

  localStorage.removeItem(
    "visitorName"
  );

};

window.addEventListener(
  "beforeunload",
  clearChat
);

return () => {

  window.removeEventListener(
    "beforeunload",
    clearChat
  );

};


}, []);

const getResponse = (question) => {


const q = question
  .toLowerCase()
  .trim();

if (
  q.includes("name")
) {
  return "My name is Sai Nikhil.";
}

if (
  q.includes("about")
) {
  return `


Sai Nikhil is a Full Stack Developer specializing in React.js, Node.js, Express.js and MongoDB.
`;
}


if (
  q.includes("skills")
) {
  return `


React.js
Node.js
Express.js
MongoDB
JavaScript
Python
Java
`;
}


if (
  q.includes("projects")
) {
  return `


1. Real-Time Sign language detection

2. Detection of stress and depression using ml

3. Job Portal

4. Portfolio Website

5. Real-Time chat Application
   `;
   }

   if (
   q.includes("certification")
   ) {
   return `
   Programming in Java

Oracle Certified Java Programming

Programming Essentials in Python

Problem Solving through Programming in C
`;
}


if (
  q.includes("github")
) {
  return `


GitHub:
https://github.com/sainikhil1909
`;
}


if (
  q.includes("linkedin") ||
  q.includes("linkdin") ||
  q.includes("linkedlin") ||
  q.includes("linked in")
) {
  return `


LinkedIn:
https://www.linkedin.com/in/sai-nikhil-palla-19042003p/
`;
}


if (
  q.includes("contact")
) {
  return `


Email:
[pallasainikhil19@gmail.com](mailto:pallasainikhil19@gmail.com)

Phone:
+91 9154638079
`;
}


if (
  q.includes("resume")
) {
  return `


Please check the Resume section of my portfolio.
`;
}


return `


Sorry ${visitorName},
I don't know that yet.
Try asking:

• About
• Skills
• Projects
• Certifications
• GitHub
• LinkedIn
• Contact
`;
};

const sendMessage = () => {


if (!input.trim()) return;

const userMessage = {

  sender: "user",

  text: input

};

/* First Question = Name */

if (!nameAsked) {

  const name = input;

  setVisitorName(name);

  setNameAsked(true);

  setMessages(prev => [

    ...prev,

    userMessage,

    {
      sender: "bot",

      text:
        `Nice to meet you ${name} 😊


I'm Sai Nikhil's Assistant.

You can ask me about:

• About
• Skills
• Projects
• Certifications
• GitHub
• LinkedIn
• Contact
• Resume`
}


  ]);

  setInput("");

  return;
}

const botMessage = {

  sender: "bot",

  text: getResponse(input)

};

setMessages(prev => [

  ...prev,

  userMessage,

  botMessage

]);

setInput("");


};

return (
<>
<button
className="chat-toggle"
onClick={() =>
setOpen(!open)
}
>
{
open
? <FaTimes />
: <FaRobot />
} </button>


  {
    open && (

      <div className="chatbot">

        <div className="chat-header">

          Sai Nikhil Assistant

        </div>

        <div className="chat-body">

          {
            messages.map(
              (msg, index) => (

                <div
                  key={index}
                  className={msg.sender}
                >
                  {msg.text}
                </div>

              )
            )
          }

          <div
            ref={messagesEndRef}
          />

        </div>

        {
          nameAsked && (

            <div className="quick-buttons">

              <button
                onClick={() =>
                  setInput("about")
                }
              >
                About
              </button>

              <button
                onClick={() =>
                  setInput("skills")
                }
              >
                Skills
              </button>

              <button
                onClick={() =>
                  setInput("projects")
                }
              >
                Projects
              </button>

              <button
                onClick={() =>
                  setInput("linkedin")
                }
              >
                LinkedIn
              </button>

            </div>

          )
        }

        <div className="chat-input">

          <input
            type="text"
            placeholder={
              nameAsked
                ? "Ask something..."
                : "Enter your name"
            }
            value={input}
            onChange={(e) =>
              setInput(
                e.target.value
              )
            }
            onKeyDown={(e) => {

              if (
                e.key === "Enter"
              ) {

                sendMessage();

              }

            }}
          />

          <button
            onClick={
              sendMessage
            }
          >
            <FaPaperPlane />
          </button>

        </div>

      </div>

    )
  }
</>


);
}

export default ChatBot;
